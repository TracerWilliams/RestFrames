141  double GDecayFrame::GenerateTwoBodyRecursive(double M,const vector<double>& M_c,
142  const TVector3& axis_par, const TVector3& axis_perp,
143  vector<TLorentzVector>& P_c)
144  {
145  TVector3 n_par = axis_par.Unit();
146  TVector3 n_perp = axis_perp.Unit();
147 
148  int N_c = M_c.size();
149 
150  double m[2];
151  m[0] = M_c[0];
152  if(N_c == 2){
153  m[1] = M_c[1];
154  } else {
155  double mO = 0.;
156  for(int i = 1; i < N_c; i++) mO += M_c[i];
157  m[1] = mO + GetRandom()*(M - m[0] - mO);
158  }
159  double Pcm = sqrt((M*M-m[0]*m[0]-m[1]*m[1])*(M*M-m[0]*m[0]-m[1]*m[1])-4.*m[0]*m[0]*m[1]*m[1])/2./M;
160  TVector3 V_c[2];
161  V_c[0] = Pcm*n_par;
162  V_c[1] = -Pcm*n_par;
163  if(m_CosDecayAngle < -1.) m_CosDecayAngle = 1.-2.*GetRandom();
164  if(m_DeltaPhiDecayPlane < 0.) m_DeltaPhiDecayPlane = 2.*acos(-1.)*GetRandom();
165 
166  for(int i = 0; i < 2; i++) V_c[i].Rotate(-acos(m_CosDecayAngle),n_perp);
167  for(int i = 0; i < 2; i++) V_c[i].Rotate(-m_DeltaPhiDecayPlane,n_par);
168  ResetDecayAngles();
169  TLorentzVector P_child[2];
170  for(int i = 0; i < 2; i++) P_child[i].SetVectM(V_c[i], m[i]);
171  P_c.push_back(P_child[0]);
172 
173  if(N_c == 2){
174  P_c.push_back(P_child[1]);
175  return Pcm;
176  }
177 
178  // Recursively generate other two-body decays
179  vector<double> M_cR;
180  for(int i = 1; i < N_c; i++) M_cR.push_back(M_c[i]);
181  TVector3 boost = P_child[1].BoostVector();
182  vector<TLorentzVector> P_cR;
183  Pcm *= GenerateTwoBodyRecursive(m[1], M_cR, boost, V_c[0].Cross(axis_par), P_cR);
184  for(int i = 0; i < N_c-1; i++) P_cR[i].Boost(boost);
185  for(int i = 0; i < N_c-1; i++) P_c.push_back(P_cR[i]);
186 
187  return Pcm;
