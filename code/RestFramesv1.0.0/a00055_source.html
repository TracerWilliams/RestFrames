43  LabGenFrame LAB_G("LAB_G","LAB");
44  DecayGenFrame Z_G("Z_G","Z");
45  VisibleGenFrame Lp_G("Lp_G","#it{l}^{+}");
46  VisibleGenFrame Lm_G("Lm_G","#it{l}^{-}");

48  LAB_G.SetChildFrame(Z_G);
49  Z_G.AddChildFrame(Lp_G);
50  Z_G.AddChildFrame(Lm_G);
51 
52  if(!LAB_G.InitializeTree()) cout << "Problem with generator tree" << endl;
53 
54  Z_G.SetMass(mZ);
55 
56  if(!LAB_G.InitializeAnalysis()) cout << "Problem with generator tree" << endl;
57 
58  LabRecoFrame LAB("LAB","LAB");
59  DecayRecoFrame Z("Z","Z");
60  VisibleRecoFrame Lp("Lp","#it{l}^{+}");
61  VisibleRecoFrame Lm("Lm","#it{l}^{-}");
62 
63  LAB.SetChildFrame(Z);
64  Z.AddChildFrame(Lp);
65  Z.AddChildFrame(Lm);
66 
67  if(!LAB.InitializeTree()) cout << "Problem with reconstruction tree" << endl;
68 
69  TreePlot* tree_plot = new TreePlot("TreePlot","TreePlot");
70 
71  // generator tree
72  tree_plot->SetTree(LAB_G);
73  tree_plot->Draw("GenTree", "Generator Tree");
74 
75  // reco tree
76  tree_plot->SetTree(LAB);
77  tree_plot->Draw("RecoTree", "Reconstruction Tree");
78 
79  if(!LAB.InitializeAnalysis()) cout << "Problem with jigsaws" << endl;
80 
81  // Declare observables for histogram booking
82  HistPlot* hist_plot = new HistPlot("HistPlot","Z #rightarrow #it{l}^{+} #it{l}^{-}");
83 
84  const HistPlotVar& MZ = hist_plot->GetNewVar("MZ", "M_{Z}", 85., 90., "[GeV]");
85  const HistPlotVar& cosZ = hist_plot->GetNewVar("cosZ","cos #theta_{Z}", -1., 1.);
86  const HistPlotVar& dphiZ = hist_plot->GetNewVar("dphiZ", "#Delta #phi_{Z}", 0., 2.*acos(-1.));
87 
88  hist_plot->AddPlot(MZ);
89  hist_plot->AddPlot(cosZ);
90  hist_plot->AddPlot(dphiZ);
91  hist_plot->AddPlot(cosZ, MZ);
92 
93  for(int igen = 0; igen < Ngen; igen++){
94  if(igen%(Ngen/10) == 0) cout << "Generating event " << igen << " of " << Ngen << endl;
95 
96  // generate event
97  LAB_G.ClearEvent(); // clear the gen tree
98  double PTZ = mZ*gRandom->Rndm();
99  LAB_G.SetTransverseMomenta(PTZ); // give the Z some Pt
100  double PzZ = mZ*(2.*gRandom->Rndm()-1.);
101  LAB_G.SetLongitudinalMomenta(PzZ); // give the Z some Pz
102  LAB_G.AnalyzeEvent(); // generate a new event
103 
104  // analyze event

106  Lp.SetLabFrameFourVector(Lp_G.GetFourVector()); // Set lepton 4-vec
107  Lm.SetLabFrameFourVector(Lm_G.GetFourVector()); // Set lepton 4-vec
108  LAB.AnalyzeEvent(); // analyze the event
109 
110  // calculate observables
111  MZ = Z.GetMass();
112  cosZ = Z.GetCosDecayAngle();
113  dphiZ = LAB.GetDeltaPhiDecayPlanes(Z);
114 
115  hist_plot->Fill();
116  }
117 
118  hist_plot->Draw();
119  hist_plot->WriteOutput(output_name);
120  hist_plot->WriteHist(output_name);
121  tree_plot->WriteOutput(output_name);
122 
123 }
